def == (obj)
  if obj.class == Literal && obj.contents == @contents && (obj.lang == @lang || (obj.lang == nil && @lang == nil))
    true
  else
    false
  end
end

def to_n3
  out = "\"" + @contents + "\""
  out += "@" + @lang if @lang != nil
  out += "^^" + @encoding if @encoding != nil
  return out
end

def to_trix
  if @lang != nil && @lang != false
    out = "<plainLiteral xml:lang=\"" + @lang + "\">"
  else
    out = "<plainLiteral>"
  end
  out += @contents
  out += "</plainLiteral>"
  return out
end
